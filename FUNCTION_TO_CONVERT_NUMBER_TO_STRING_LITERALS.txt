FUNCTION TO CONVERT NUMBERS TO STRING USING JULIAN DATE FORMAT

FOR EXAMPLE :

OUR SOURCE IS 

OCCURS

5

--FIRING FUNCTION
SELECT OCCURS ,FUNC(OCCURS) STRING
FROM SCENE9;

OCCURS	STRING

8	EIGHT

*/


--------CREATING A FUNCTION OF SAME PROC
CREATE OR REPLACE FUNCTION FUNC
( ARG1 IN NUMBER) RETURN NVARCHAR2 AS ARG2  NVARCHAR2(1000);
BEGIN
SELECT TO_CHAR(TO_dATE(ARG1,'J'),'JSP') INTO ARG2 FROM DUAL;
RETURN ARG2;
END ;
